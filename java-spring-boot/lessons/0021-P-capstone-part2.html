Step 1：Category Entity（一的一方）

category/Category.java：

package com.blogapi.category;

import com.blogapi.post.Post;
import jakarta.persistence.*;
import java.util.*;

@Entity
@Table(name = "categories")
public class Category {

    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(nullable = false, unique = true, length = 50)
    private String name;

    @Column(unique = true, length = 60)
    private String slug;   // URL-friendly name

    @OneToMany(mappedBy = "category")
    private List<Post> posts = new ArrayList<>();

    protected Category() { }

    public Category(String name, String slug) {
        this.name = name;
        this.slug = slug;
    }

    public Long getId() { return id; }
    public String getName() { return name; }
    public String getSlug() { return slug; }
    public List<Post> getPosts() { return posts; }
}

category/CategoryService.java：

package com.blogapi.category;

import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
public class CategoryService {
    private final CategoryRepository categoryRepo;

    public CategoryService(CategoryRepository categoryRepo) {
        this.categoryRepo = categoryRepo;
    }

    public Category getOrCreate(String name) {
        return categoryRepo.findByName(name)
            .orElseGet(() -> categoryRepo.save(
                new Category(name, toSlug(name))));
    }

    public Category getBySlug(String slug) {
        return categoryRepo.findBySlug(slug)
            .orElseThrow(() -> new IllegalArgumentException("分類不存在: " + slug));
    }

    private String toSlug(String name) {
        return name.toLowerCase().replaceAll("[^a-z0-9\\u4e00-\\u9fff]+", "-")
            .replaceAll("^-|-$", "");
    }
}

Step 2：Post Entity（多的一方）

post/Post.java：

package com.blogapi.post;

import com.blogapi.category.Category;
import com.blogapi.user.User;
import jakarta.persistence.*;
import java.time.LocalDateTime;

@Entity
@Table(name = "posts")
public class Post {

    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(nullable = false, length = 200)
    private String title;

    @Column(unique = true, length = 250)
    private String slug;

    @Column(nullable = false, columnDefinition = "TEXT")
    private String content;

    @Column(length = 500)
    private String summary;

    @Column(nullable = false)
    @Enumerated(EnumType.STRING)
    private PostStatus status = PostStatus.DRAFT;

    @Column(name = "created_at", nullable = false)
    private LocalDateTime createdAt;

    @Column(name = "updated_at")
    private LocalDateTime updatedAt;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "author_id", nullable = false)
    private User author;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "category_id")
    private Category category;

    public enum PostStatus { DRAFT, PUBLISHED }

    protected Post() { }

    public Post(String title, String content, User author, Category category) {
        this.title = title;
        this.slug = generateSlug(title);
        this.content = content;
        this.author = author;
        this.category = category;
        this.status = PostStatus.DRAFT;
        this.createdAt = LocalDateTime.now();
    }

    // === 業務方法 ===
    public void publish() {
        this.status = PostStatus.PUBLISHED;
        this.updatedAt = LocalDateTime.now();
    }

    public void update(String title, String content, String summary, Category category) {
        if (title != null) { this.title = title; this.slug = generateSlug(title); }
        if (content != null) this.content = content;
        if (summary != null) this.summary = summary;
        if (category != null) this.category = category;
        this.updatedAt = LocalDateTime.now();
    }

    // === Getters ===
    public Long getId() { return id; }
    public String getTitle() { return title; }
    public String getSlug() { return slug; }
    public String getContent() { return content; }
    public String getSummary() { return summary; }
    public PostStatus getStatus() { return status; }
    public LocalDateTime getCreatedAt() { return createdAt; }
    public LocalDateTime getUpdatedAt() { return updatedAt; }
    public User getAuthor() { return author; }
    public Category getCategory() { return category; }

    // === Slug 產生（簡單版：title + 時間戳） ===
    private String generateSlug(String title) {
        String base = title.toLowerCase()
            .replaceAll("[^a-z0-9\\u4e00-\\u9fff]+", "-")
            .replaceAll("^-|-$", "");
        return base + "-" + System.currentTimeMillis() % 100000;
    }
}
💡 @Enumerated(EnumType.STRING) 這告訴 JPA 把 enum 存成字串（DRAFT / PUBLISHED）而非數字。數字（ORDINAL）在 enum 順序改變時會造成資料錯亂——永遠用 STRING。

post/dto/PostResponse.java（含巢狀 author + category 摘要）：

package com.blogapi.post.dto;

import com.blogapi.post.Post;
import java.time.LocalDateTime;

public record PostResponse(
    Long id, String title, String slug, String content,
    String summary, String status, LocalDateTime createdAt,
    LocalDateTime updatedAt,
    AuthorInfo author, CategoryInfo category
) {
    public record AuthorInfo(Long id, String username, String displayName) { }
    public record CategoryInfo(Long id, String name, String slug) { }

    public static PostResponse from(Post post) {
        var author = new AuthorInfo(
            post.getAuthor().getId(), post.getAuthor().getUsername(),
            post.getAuthor().getDisplayName());
        var cat = post.getCategory() != null
            ? new CategoryInfo(post.getCategory().getId(),
                post.getCategory().getName(), post.getCategory().getSlug())
            : null;
        return new PostResponse(post.getId(), post.getTitle(), post.getSlug(),
            post.getContent(), post.getSummary(),
            post.getStatus().name(), post.getCreatedAt(),
            post.getUpdatedAt(), author, cat);
    }
}

post/PostService.java：

package com.blogapi.post;

import com.blogapi.category.CategoryService;
import com.blogapi.post.dto.*;
import com.blogapi.user.UserRepository;
import org.springframework.data.domain.*;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
public class PostService {
    private final PostRepository postRepo;
    private final UserRepository userRepo;
    private final CategoryService categoryService;

    public PostService(PostRepository postRepo, UserRepository userRepo,
                       CategoryService categoryService) {
        this.postRepo = postRepo;
        this.userRepo = userRepo;
        this.categoryService = categoryService;
    }

    @Transactional
    public PostResponse create(CreatePostRequest req) {
        var author = userRepo.findById(req.authorId())
            .orElseThrow(() -> new IllegalArgumentException("作者不存在: " + req.authorId()));

        var category = req.category() != null
            ? categoryService.getOrCreate(req.category()) : null;

        var post = new Post(req.title(), req.content(), author, category);
        if (req.summary() != null) post.update(null, null, req.summary(), null);

        return PostResponse.from(postRepo.save(post));
    }

    @Transactional
    public PostResponse publish(Long postId) {
        var post = postRepo.findById(postId)
            .orElseThrow(() -> new IllegalArgumentException("文章不存在: " + postId));
        post.publish();
        return PostResponse.from(postRepo.save(post));
    }

    @Transactional
    public PostResponse update(Long postId, UpdatePostRequest req) {
        var post = postRepo.findById(postId)
            .orElseThrow(() -> new IllegalArgumentException("文章不存在: " + postId));

        var category = req.category() != null
            ? categoryService.getOrCreate(req.category()) : null;

        post.update(req.title(), req.content(), req.summary(), category);
        return PostResponse.from(postRepo.save(post));
    }

    public PostResponse getBySlug(String slug) {
        return PostResponse.from(
            postRepo.findBySlug(slug)
                .orElseThrow(() -> new IllegalArgumentException("文章不存在: " + slug))
        );
    }

    public PostResponse getById(Long id) {
        return PostResponse.from(
            postRepo.findById(id)
                .orElseThrow(() -> new IllegalArgumentException("文章不存在: " + id))
        );
    }

    public Page<PostResponse> listPublished(int page, int size) {
        return postRepo.findByStatus(Post.PostStatus.PUBLISHED,
                PageRequest.of(page, size, Sort.by("createdAt").descending()))
            .map(PostResponse::from);
    }

    public Page<PostResponse> search(String keyword, int page, int size) {
        return postRepo.findByTitleContainingIgnoreCase(keyword,
                PageRequest.of(page, size, Sort.by("createdAt").descending()))
            .map(PostResponse::from);
    }

    @Transactional
    public void delete(Long postId) {
        postRepo.deleteById(postId);
    }
}
🔍 設計觀察 PostService 依賴 CategoryService 來處理分類——getOrCreate() 自動建立不存在的分類。這展示了 Service 之間的正確協作方式：PostService 不直接碰 CategoryRepository，而是透過 CategoryService 的公共介面。

Step 5：Controller

post/PostController.java：

package com.blogapi.post;

import com.blogapi.common.ApiResponse;
import com.blogapi.post.dto.*;
import jakarta.validation.Valid;
import org.springframework.data.domain.Page;
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/posts")
public class PostController {
    private final PostService postService;

    public PostController(PostService postService) {
        this.postService = postService;
    }

    // 公開：已發佈文章列表（分頁）
    @GetMapping
    public ApiResponse<Page<PostResponse>> list(
        @RequestParam(defaultValue = "0") int page,
        @RequestParam(defaultValue = "10") int size
    ) {
        return ApiResponse.ok(postService.listPublished(page, size));
    }

    // 搜尋
    @GetMapping("/search")
    public ApiResponse<Page<PostResponse>> search(
        @RequestParam String q,
        @RequestParam(defaultValue = "0") int page,
        @RequestParam(defaultValue = "10") int size
    ) {
        return ApiResponse.ok(postService.search(q, page, size));
    }

    // 依 slug 查詢
    @GetMapping("/slug/{slug}")
    public ApiResponse<PostResponse> getBySlug(@PathVariable String slug) {
        return ApiResponse.ok(postService.getBySlug(slug));
    }

    // 依 ID 查詢
    @GetMapping("/{id}")
    public ApiResponse<PostResponse> getById(@PathVariable Long id) {
        return ApiResponse.ok(postService.getById(id));
    }

    // 建立
    @PostMapping
    @ResponseStatus(HttpStatus.CREATED)
    public ApiResponse<PostResponse> create(@Valid @RequestBody CreatePostRequest req) {
        return ApiResponse.ok(postService.create(req));
    }

    // 發佈
    @PatchMapping("/{id}/publish")
    public ApiResponse<PostResponse> publish(@PathVariable Long id) {
        return ApiResponse.ok(postService.publish(id));
    }

    // 更新
    @PutMapping("/{id}")
    public ApiResponse<PostResponse> update(
        @PathVariable Long id,
        @Valid @RequestBody UpdatePostRequest req
    ) {
        return ApiResponse.ok(postService.update(id, req));
    }

    // 刪除
    @DeleteMapping("/{id}")
    @ResponseStatus(HttpStatus.NO_CONTENT)
    public void delete(@PathVariable Long id) {
        postService.delete(id);
    }
}

Part 2 檢查清單

  • Category Entity + Repository + Service，支援 getOrCreate
  • Post Entity 含 @ManyToOne（author + category）+ @Enumerated(STRING)
  • CreatePostRequest + UpdatePostRequest 含 @Valid 驗證
  • PostResponse 含巢狀 AuthorInfo + CategoryInfo（不含密碼）
  • PostRepository 含分頁 + 搜尋 + 依 slug/status/authorId/categoryId 查詢
  • PostController 完整 CRUD：GET list / GET by slug / POST / PUT / PATCH publish / DELETE
  • 分頁 JSON 含 content + totalElements + totalPages
  • 所有 API 回應包在 ApiResponse 中
